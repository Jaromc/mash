251  void _Reserve(uint32 count)
252  {
253  if (count <= INTERNAL_POOL_SIZE)
254  {
255  MASH_ASSERT(m_reservedElements <= INTERNAL_POOL_SIZE);

257  m_string = &m_internalPool[0];
258  m_reservedElements = INTERNAL_POOL_SIZE;
259  }
260  else

264  if (m_reservedElements <= INTERNAL_POOL_SIZE)
265  freeOldData = false;

268 
269  MASH_ASSERT(newMem);

272  newMem[0] = 0;
