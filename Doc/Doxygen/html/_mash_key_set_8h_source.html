160  {
161  *out = 0;
162  int32 startKey = -1;
163  int32 endKey = -1;
164  uint32 frame = 0;
165 
166  /*
167  Get bounding keys around the clip
168  */
169  T keyToFind;
170  keyToFind.frame = clip.start;
171 
172  typename MashArray<T>::ConstIterator startFrameIter = mash::algorithms::LowerBoundPred<typename MashArray<T>::ConstIterator, T, sCompareKeyFramesPred, algorithms::sPassThrough<bool> >(m_keyList.Begin(), m_keyList.End(), keyToFind, sCompareKeyFramesPred(), algorithms::sPassThrough<bool>());
173  if (startFrameIter != m_keyList.End())
174  startKey = m_keyList.Begin().Distance(startFrameIter);
175 
176  keyToFind.frame = clip.end;
177  typename MashArray<T>::ConstIterator endFrameIter = mash::algorithms::LowerBoundPred<typename MashArray<T>::ConstIterator, T, sCompareKeyFramesGreaterPred, algorithms::sPassThrough<bool> >(m_keyList.Begin(), m_keyList.End(), keyToFind, sCompareKeyFramesGreaterPred(), algorithms::sPassThrough<bool>());
178  if (endFrameIter != m_keyList.End())
179  endKey = m_keyList.Begin().Distance(endFrameIter);
180 
181  if (startKey != -1 && endKey != -1)
182  {
183  if (endKey > startKey)
184  {
185  MASH_WRITE_TO_LOG(MashLog::aERROR_LEVEL_ERROR,
186  "Animation clip start frame must be less than the end frame.",
187  "MashTransformationKeySet::Chop");
188 
189  return aMASH_FAILED;
190  }
191 
192  MashKeySet<T> *newKeySet = (MashKeySet<T>*)CreateNewSet();
193  if (!newKeySet)
194  return aMASH_FAILED;
195 
196  int32 frameAdjustmentTime = -m_keyList[startKey].frame;
197  if (newKeySet->AddKeyList(&m_keyList[startKey], endKey - startKey) == aMASH_FAILED)
198  {

201  }

204  The new animations time is adjusted so that it starts at 0
205  */

214  /*
215  This function uses the result of the last lookup to predict the key locations.
216  If the cached data doesn't work then a full search is performed (This will usually only
217  happen in the first call).
218  */
219  void GetFrameBoundsCached(uint32 frame, bool playingInReverse, uint32 lastFromKey, uint32 lastToKey,
220  uint32 &fromKeyOut, uint32 &toKeyOut)const
221  {
222  fromKeyOut = 0;
223  toKeyOut = 0;
224 
225  const uint32 keyCount = m_keyList.Size();
226 
227  if (keyCount < 2)
228  return;
229 
230  if (!playingInReverse)
231  {
232  //first check cached result
233  if ((m_keyList[lastFromKey].frame <= frame) && (m_keyList[lastToKey].frame >= frame))
234  {
235  fromKeyOut = lastFromKey;
236  toKeyOut = lastToKey;
237  return;
238  }
239 
240  //Before checking the next set, make sure we haven't hit the end of the keys.
241  int32 nextPredKey = lastToKey + 1;
242  if (nextPredKey >= keyCount)
243  {
244  fromKeyOut = 0;
245  toKeyOut = 1;
246  return;
247  }
248 
249  //try the next set along
250  if ((m_keyList[lastToKey].frame <= frame) && (m_keyList[nextPredKey].frame >= frame))
251  {
252  fromKeyOut = lastToKey;
253  toKeyOut = nextPredKey;
254  return;
255  }
256  }
257  else
258  {
259  //first check cached result
260  if ((m_keyList[lastFromKey].frame >= frame) && (m_keyList[lastToKey].frame <= frame))
261  {
262  fromKeyOut = lastFromKey;
263  toKeyOut = lastToKey;
264  return;
265  }
266 
267  //Before checking the next set, make sure we haven't hit the end of the keys.
268  int32 nextPredKey = lastToKey - 1;
269  if (nextPredKey < 0)
270  {
271  fromKeyOut = keyCount-1;
272  toKeyOut = math::Max<int32>(0, fromKeyOut-1);
273  return;
274  }
275 
276  //try the next set along
277  if ((m_keyList[lastToKey].frame >= frame) && (m_keyList[nextPredKey].frame <= frame))
278  {
279  fromKeyOut = lastToKey;
280  toKeyOut = nextPredKey;
281  return;
282  }
283  }
284 
285  //cant find it so search the whole array
286  GetFrameBounds(frame, fromKeyOut, toKeyOut);
287  }

316  eMASH_STATUS AddKey(const T &key)
317  {
318  /*
319  If the last key is less than the new key then we simply push back
320  the new key. Otherwise we need to find the location to insert the new key.
321  */
322  if (m_keyList.Empty() || (key.frame > m_keyList.Back().frame))
323  {
324  m_keyList.PushBack(key);
325  return aMASH_OK;
326  }
327 
328  typename MashArray<T>::Iterator upperBoundIter = mash::algorithms::LowerBoundPred<typename MashArray<T>::Iterator, T, sCompareKeyFramesGreaterPred, algorithms::sPassThrough<bool> >(m_keyList.Begin(), m_keyList.End(), key, sCompareKeyFramesGreaterPred(), algorithms::sPassThrough<bool>());
329  if (upperBoundIter != m_keyList.End())
330  {
331  //make sure frames numbers are different
332  if (upperBoundIter->frame == key.frame)
333  {
334  MASH_WRITE_TO_LOG(MashLog::aERROR_LEVEL_ERROR,
335  "Animation keys in the same set cannot share frame numbers.",
336  "MashTransformationKeySet::AddKey");
337 
338  return aMASH_FAILED;
339  }
340  }
341 
342  m_keyList.Insert(upperBoundIter, key);
343 
344  return aMASH_OK;
